library(tidyverse)
library(incur)
double_data <- incur::double_data

ggplot() +
  geom_point(
    double_data,
    mapping = aes(hours_from, area, colour = as.factor(cells_seeded))
  )

double_model <- incur::fit_model(
  data = double_data,
  x_var = "hours_from",
  y_var = "area",
  model = "five_param_sigmoid",
  share_group = "cells_seeded",
  share_params = c("top", "slope")
)

conc_data <- incur::conc_data

ggplot() +
  geom_point(
    conc_data,
    mapping = aes(hours_from, area, colour = as.factor(concentration))
  )

conc_data <- mutate(
  conc_data,
  exclude = case_when(
    concentration %in% c(-4, -4.5) ~ TRUE,
    .default = FALSE
  )
)
model_list <- incur:::fit_model_conc(
  data = conc_data,
  x_var = "hours_from",
  y_var = "area",
  model = "five_param_sigmoid",
  rout = TRUE
)

incur:::plot_model_conc(
  model_list = model_list,
  x_var = "hours_from",
  y_var = "area",
  start_colour = "#0085ca"
)

incur:::plot_model_conc_gr_time(
  model_list = model_list,
  x_var = "hours_from",
  y_var = "area",
  start_colour = "#0085ca"
)
